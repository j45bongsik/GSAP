trigger 스크롤 내릴시 애니메이션 실행할 요소가 영역에 도달하면 애니메이션 실행 
                        gsap.to(box2,{
                            duration: 2,
                            x: 500, 
                            rotation: 360,
                            borderRadius: 100,
                            scrollTrigger: {
                                trigger: box2, // 요소의 시작점 설정
                            }
                        });
toggleActions 스크롤 내릴시 : 시작점을 지날때(onEnter) / 종료점을 지날때(onLeave)
스크롤을 올릴때 : 종료점을 지날때(onEnterBack) / 시작점을 지날때 (onLeaveBack)
애니메이션 행동을 설정한다. play, pause, resume, reset, restart, complete, reverse, none 요소 값을 설정 
                        gsap.to(box3,{
                            duration: 1,
                            x:500,
                            rotation: 360,
                            borderRadius: 100,
                            scrollTrigger: {
                                trigger: box3, // 요소의 시작점 설정
                                toggleActions: "play pause reverse none", // 순서대로 onEnter / onLeave / onEnterBack / onLeaveBack
                            }
                        });
start, end 에니메이션을 실행시킬 시작점과 종료지점을 설정 
                        gsap.to(box4,{
                            duration: 1,
                            x:500, 
                            rotation: 360,
                            borderRadius: 100,
                            scrollTrigger: {
                                trigger: box4, // 요소의 시작점 설정
                                start: "top 50%", // 시작점
                                end: "bottom 20%", // 종료점
                                toggleActions: "play reverse restart pause",// 순서대로 onEnter / onLeave / onEnterBack / onLeaveBack
                                markers: true, // 마커의 위치 확인
                            }
                        })
scrub 마우스 스크롤 위치에 따라 애니메이션 따라옴 
                        gsap.to(box5,{
                            duration: 1,
                            x: 500,
                            rotation: 360,
                            scrollTrigger: {
                                trigger: box5, // 요소의 시작점 설정
                                start: "top 60%", // 시작점
                                end: "bottom 20%", // 종료점
                                scrub: 5, // 마우스 스크롤 값에 따른 애니메이션 설정 (true 및 정수)
                                markers: true, // 마커의 위치 확인
                            }
                        });
pin 애니메이션 실행하다가 지정한 종료점에 도달시 지정한 애니메이션 종료값으로 고정시킨다. 
                        gsap.to(box6,{
                            duration: 2,
                            x: 500,
                            rotation: 360,
                            borderRadius: 100,
                            scrollTrigger: {
                                trigger: box6, // 요소의 시작점 설정
                                start: "top 50%", // 시작점
                                end: "top 400px", // 종료점
                                scrub: true, // 마우스 스크롤 값에 따른 애니메이션 설정 (true 및 정수)
                                markers: true, // 마커의 위치 확인
                                pin: true, // 고정시키기 위해 true
                            }
                        });
toggleClass 시작점에 도달하면 클래스를 추가시키고 종료점에 도착하면 클래스를 삭제한다. 
                        gsap.to(box7,{
                            duration: 2,
                            x: 500,
                            rotation: 360,
                            borderRadius: 100,
                            scrollTrigger: {
                                trigger: box7, // 요소의 시작점 설정
                                start: "top center", // 시작점
                                end: "bottom top", // 종료점
                                scrub: true,  // 마우스 스크롤 값에 따른 애니메이션 설정 (true 및 정수)
                                markers: true, // 마커의 위치 확인
                                toggleClass: "active", // 추가할 클래스
                                id: "box7" //마커의 이름을 변경
                            }
                        });
callback 시작점과 종료점을 지날때마다 콜백 함수가 제공된다

                            }
                        });
